from gpiozero import LED

led1 = LED(24)
sensor = DistanceSensor(echo=18, trigger=25,pin_factory=factory)
servo = Servo(12,min_pulse_width=0.5/1000,max_pulse_width=2.5/1000,pin_factory=factory)
def forward():
    led.on()
    led1.on()
    print('LED ON')
    sleep(5)
    led.off()
    led1.off()

def left():

 for number in range(5):
